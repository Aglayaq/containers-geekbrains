Применяем deployment.yaml

--- 

apiVersion: apps/v1
kind: Deployment
metadata:
  name: fact-deployment
spec:
  replicas: 1
  strategy:
   type: RollingUpdate
   rollingUpdate:
    maxSurge: 1
    maxUnavailable: 0        
  selector:
    matchLabels:
      app: my-app
  template:
    metadata:
      labels:
        app: my-app
    spec:
      containers:
      - image: factoriotools/factorio:0.18.34
        name: factorio
        ports:
        - containerPort: 34197
          protocol: UDP
        - containerPort: 27015
          protocol: TCP
